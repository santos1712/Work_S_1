from math import sin, cos
from turtle import *

def hearta(K):
    return 15 * sin(K) ** 3

def heartb(K):
    return 12 * cos(K) - 5 * cos(2 * K) - 2 * cos(3 * K) - cos(4 * K)

speed(0)
bgcolor("black")
color("#f73487", "#f73487")
penup()

begin_fill()

K = 0
while K < 2 * 3.1416:
    x = hearta(K) * 20
    y = heartb(K) * 20
    goto(x, y)
    pendown()
    K += 0.01

end_fill()
done()
